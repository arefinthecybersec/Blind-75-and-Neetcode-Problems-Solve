4. Find the missing number in the array:

numbers[] = {0, 1, 2, 4, 5}
Use the formula: n(n+1)/2 - sum(array)

âœ… Output:3

======================================

#include<bits/stdc++.h>
using namespace std;

int arrSum(vector<int> &numbers) {
    int sum = 0;
    for(auto u: numbers) {
        sum += u;
    }
    return sum;
}

void missingNum(vector<int> &numbers) {
    int n = numbers.size();
    int ans = n*(n+1)/2;
    cout << ans - arrSum(numbers);
}

int main() {
    vector<int> numbers = {0, 1, 2, 3, 5, 6};

    missingNum(numbers);
    return 0;
}
