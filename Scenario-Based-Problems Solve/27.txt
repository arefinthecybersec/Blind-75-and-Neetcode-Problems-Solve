using namespace std;

int main() {
    string s = "cat";
    cout << s << endl;

Approach: 2 (use built-in func) - Array Method

#include <bits/stdc++.h>
using namespace std;

void arrPermute(int arr[], int n) {
    // initial permuation: 
    for(int i = 0; i < n; i++) {
        cout << arr[i];
    }
    cout << endl;

    // making permuation:
    while(next_permutation(arr, arr+n)) {
        for(int i = 0; i < n; i++) {
            cout << arr[i];
        }
        cout << endl;
    }
}

int main() {
    int arr[] {1, 2, 3};
    int n = sizeof(arr)/sizeof(arr[0]);
    sort(arr, arr+n);

    arrPermute(arr, n);
    return 0;
}
